NAME:
   tdbg taskqueue describe-task-queue-partition - Describe information related to a task queue partition

USAGE:
   tdbg taskqueue describe-task-queue-partition [command options]

OPTIONS:
   --namespace-id value                                 NamespaceId (default: "default")
   --task-queue value                                   Task Queue name
   --task-queue-type value                              Task Queue type: activity, workflow, nexus (experimental) (default: "TASK_QUEUE_TYPE_WORKFLOW")
   --partition-id value                                 Partition ID (default: 0)
   --sticky-name value                                  Sticky Name for a task queue partition, if present
   --select-build-id value [ --select-build-id value ]  Build IDs
   --select-unversioned                                 Unversioned task queue partition (default: true)
   --select-all-active                                  All active task queue versions (default: true)
   --help, -h                                           show help

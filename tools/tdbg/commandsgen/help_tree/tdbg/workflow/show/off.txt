NAME:
   tdbg workflow show - show workflow history from database

USAGE:
   tdbg workflow show [command options]

OPTIONS:
   --workflow-id value, --wid value  Workflow ID
   --run-id value, --rid value       Run ID
   --min-event-id value              Minimum event ID to be included in the history (default: 0)
   --max-event-id value              Maximum event ID to be included in the history (default: 9223372036854775807)
   --min-event-version value         Start event version to be included in the history (default: 0)
   --max-event-version value         End event version to be included in the history (default: 0)
   --output-filename value           output file
   --help, -h                        show help
